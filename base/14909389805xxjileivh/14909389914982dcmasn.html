// –ù–µ—Å–∫–æ–ª—å–∫–æ –º–æ–¥–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –ø—Ä–∏–º–µ—Ä Example 13-1. 
// –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ K-means
//
// –∏–∑ –∫–Ω–∏–≥–∏:
//   Learning OpenCV: Computer Vision with the OpenCV Library
//     by Gary Bradski and Adrian Kaehler
//     Published by O'Reilly Media, October 3, 2008

#include "cxcore.h"
#include "highgui.h"

#define MAX_CLUSTERS 5

int main( int argc, char* argv[] )
{
	// –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –¥–ª—è –ø–æ–∫–∞–∑–∞ —Ç–æ—á–µ–∫ 
	IplImage* img = cvCreateImage( cvSize( 500, 500 ), 8, 3 );

	// —Ç–∞–±–ª–∏—Ü–∞ —Ü–≤–µ—Ç–æ–≤ –∫–ª–∞—Å—Ç–µ—Ä–æ–≤
	CvScalar color_tab[MAX_CLUSTERS];
	color_tab[0] = CV_RGB(255,0,0);
	color_tab[1] = CV_RGB(0,255,0);
	color_tab[2] = CV_RGB(100,100,255);
	color_tab[3] = CV_RGB(255,0,255);
	color_tab[4] = CV_RGB(255,255,0);
	
	// –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Å–æ—Å—Ç–æ—è–Ω–∏—è –ì–ü–°–ß
	CvRNG rng = cvRNG(0xffffffff);

	cvNamedWindow( "clusters", 1 );

	for(;;){
		int k, cluster_count = cvRandInt(&rng)%MAX_CLUSTERS + 1;
		int i, sample_count = cvRandInt(&rng)%1000 + 1;
		CvMat* points = cvCreateMat( sample_count, 1, CV_32FC2 );
		CvMat* clusters = cvCreateMat( sample_count, 1, CV_32SC1 );

		// –≥–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–ª—É—á–∞–π–Ω–æ–≥–æ –≥–∞—É—Å—Å–æ–≤–æ–≥–æ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è —Ç–æ—á–µ–∫
		for( k = 0; k < cluster_count; k++ ){
			CvPoint center;
			CvMat point_chunk;
			center.x = cvRandInt(&rng)%img->width;
			center.y = cvRandInt(&rng)%img->height;
			cvGetRows( points, &point_chunk, 
				k*sample_count/cluster_count,
				k == cluster_count - 1 ? sample_count :  
				(k+1)*sample_count/cluster_count );
			cvRandArr( &rng, &point_chunk, CV_RAND_NORMAL,
				cvScalar(center.x,center.y,0,0),
				cvScalar(img->width/6, img->height/6,0,0) );
		}

		// —Ç–æ—á–∫–∏ –ø–µ—Ä–µ–º–µ—à–∏–≤–∞—é—Ç—Å—è
		for( i = 0; i < sample_count/2; i++ ){
			CvPoint2D32f* pt1 = (CvPoint2D32f*)points->data.fl +
				cvRandInt(&rng)%sample_count;
			CvPoint2D32f* pt2 = (CvPoint2D32f*)points->data.fl + 
				cvRandInt(&rng)%sample_count;
			CvPoint2D32f temp;
			CV_SWAP( *pt1, *pt2, temp );
		}

		// –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∫–ª–∞—Å—Ç–µ—Ä–æ–≤
		cvKMeans2( points, cluster_count, clusters,
			cvTermCriteria( CV_TERMCRIT_EPS+CV_TERMCRIT_ITER, 
			10, 1.0 ));
		cvZero( img );

		// –ø–æ–∫–∞–∑—ã–≤–∞–µ–º —Ç–æ—á–∫–∏
		for( i = 0; i < sample_count; i++ ){
			CvPoint2D32f pt = ((CvPoint2D32f*)points->data.fl)[i];
			// –∏–Ω–¥–µ–∫—Å –∫–ª–∞—Å—Ç–µ—Ä–∞
			int cluster_idx = clusters->data.i[i];
			cvCircle( img, cvPointFrom32f(pt), 2, color_tab[cluster_idx], CV_FILLED );
		}

		cvReleaseMat( &points );
		cvReleaseMat( &clusters );

		// –ø–æ–∫–∞–∑—ã–≤–∞–µ–º
		cvShowImage( "clusters", img );

		int key = cvWaitKey(330);
		if( key == 27 ){ // 'ESC'
			break;
		}
	}
	
	// –æ—Å–≤–æ–±–æ–∂–¥–∞–µ–º —Ä–µ—Å—É—Ä—Å—ã
	cvReleaseImage(&img);
	// —É–¥–∞–ª—è–µ–º –æ–∫–Ω–∞
	cvDestroyAllWindows();
	return 0;
}
